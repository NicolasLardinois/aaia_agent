```python

from agents.stock_deep_dive.index.index_valuation_range_agent import _combine
from core.domain.models import Signal

def test_one_bullish_one_neutral_is_neutral_not_bullish():
    """Nur eine Methode zeigt undervalued — zu unsicher für BULLISH."""
    pos, sig = _combine(m1_pts=1, m2_pts=0)
    assert sig == Signal.NEUTRAL

def test_both_bullish_is_bullish():
    pos, sig = _combine(m1_pts=1, m2_pts=1)
    assert sig == Signal.BULLISH

def test_both_bearish_is_bearish():
    pos, sig = _combine(m1_pts=-1, m2_pts=-1)
    assert sig == Signal.BEARISH

def test_one_bearish_one_neutral_is_neutral():
    pos, sig = _combine(m1_pts=-1, m2_pts=0)
    assert sig == Signal.NEUTRAL
```

- [ ] **Step 2: Tests ausführen und Fehler bestätigen**

```
pytest tests/agents/stock_deep_dive/index/test_index_valuation_range.py::test_one_bullish_one_neutral_is_neutral_not_bullish -v
```
Expected: FAIL (ist BULLISH)

- [ ] **Step 3: Fix — Schwelle auf 2 erhöhen**

```python
def _combine(m1_pts: int, m2_pts: int) -> tuple[str, Signal]:
    total = m1_pts + m2_pts
    if total >= 2:        # war: >= 1 — jetzt beide Methoden müssen einig sein
        return "undervalued", Signal.BULLISH
    if total <= -2:       # war: <= -1
        return "overvalued", Signal.BEARISH
    return "fair", Signal.NEUTRAL
```
